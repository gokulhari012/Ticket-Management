sudo nano /etc/systemd/system/flaskapp.service

----------------------------------------------
[Unit]
Description=Flask Web Application
After=network.target

[Service]
User=pi
# User=root               # Run as root (sudo)
WorkingDirectory=/home/pi/myflaskapp  # Replace with your Flask app's directory
ExecStart=/usr/bin/python3 /home/pi/myflaskapp/app.py  # Replace with your app's entry point
Restart=always
# Environment=PATH=/usr/bin:/usr/local/bin  # Ensure that the correct PATH is used

[Install]
WantedBy=multi-user.target

----------------------------------------------

sudo systemctl daemon-reload
sudo systemctl enable flaskapp

sudo systemctl start flaskapp
